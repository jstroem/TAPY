\chapter{Complete CFG node reference}
\label{chapter:NodeRef}

\begin{description}
	\item[FunctionDeclNode \inlinecode{def f(...):}] Function declarations.
	\item[ClassDeclNode \inlinecode{class C: ...}] Class declarations.
	\item[ModuleEntryNode] Module entry.
	\item[ClassEntryNode] Class entry.
	\item[FunctionEntryNode] Function entry.
	\item[ClassExitNode] Class exit.
	\item[FunctionExitNode] Function exit.
	\item[ExceptionalExitNode] Exceptional function exit.
	\item[WriteVariableNode \inlinecode{x = reg}] Writing to variables.
	\item[WriteRegisterNode \inlinecode{reg$_{\inlinecode{to}}$ = reg$_{\inlinecode{from}}$}] Writing/copying to register.
	\item[WriteAttributeNode \inlinecode{reg$_{\inlinecode{obj}}$.attr = reg$_{\inlinecode{from}}$}] Writing to attributes of objects.
	\item[WriteIndexableNode \inlinecode{reg$_{\inlinecode{obj}}$[reg$_{\inlinecode{index}}$] = reg$_{\inlinecode{from}}$}] Writing to items of objects.
	\item[ConstantBooleanNode] Boolean values.
	\item[ConstantIntNode] Integer values.
	\item[ConstantFloatNode] Float values.
	\item[ConstantLongNode] Long values.
	\item[ConstantComplexNode] Complex number values.
	\item[ConstantStringNode] String values.
	\item[ConstantNoneNode] Built-in constant \inlinecode{None}.
	\item[ReadVariableNode \inlinecode{x}] Reading variables.
	\item[ReadAttributeNode \inlinecode{reg$_{\inlinecode{obj}}$.attr}] Reading attributes of objects.
	\item[ReadIndexableNode \inlinecode{reg$_{\inlinecode{obj}}$[reg$_{\inlinecode{index}}$]}] Reading items of objects.
	\item[HasAttributeNode \inlinecode{hasattr(reg$_{\inlinecode{obj}}$, 'attr')}] Feature detection.
%	\item[DelVariableNode \inlinecode{del x}] Introduced when deleting a variable from the current scope.
%	\item[DelVariableNode \inlinecode{del reg$_{\inlinecode{obj}}$.attr}] Introduced when deleting an attribute on an object.
%	\item[DelIndexableNode \inlinecode{reg$_{\inlinecode{obj}}$[reg$_{\inlinecode{index}}$]}] Introduced when deleting a indexable attribute on an object.
	\item[NoOpNode \inlinecode{pass}] No operation node.
	\item[BreakNode \inlinecode{break}] Loop breaking.
	\item[ContinueNode \inlinecode{break}] Loop continuing.
	\item[ReturnNode \inlinecode{return reg$_{\inlinecode{value}}$}] Function return.
	\item[IfNode \inlinecode{if reg$_\inlinecode{cond}$:}] \inlinecode{if} statements.
	\item[CallNode \inlinecode{reg$_{\inlinecode{function}}$(reg$_{\inlinecode{1}}$, ..., reg$_{\inlinecode{n}}$)}] Function calls.
	\item[AfterCallNode] Stores the result of a call.
	\item[RaiseNode \inlinecode{raise reg$_\inlinecode{exception}$}] Raising exceptions.
	\item[ExceptNode \inlinecode{except (type$_\inlinecode{1}$, ..., type$_\inlinecode{n}$)}:] Catching exceptions.
	\item[CompareOpNode \inlinecode{reg$_\inlinecode{left}$ op$_\inlinecode{comp}$ reg$_\inlinecode{right}$}] Compare expressions.
	\item[BinOpNode \inlinecode{reg$_\inlinecode{left}$ op$_\inlinecode{binop}$ reg$_\inlinecode{right}$}] Binary operator expressions.
	\item[UnOpNode \inlinecode{op reg$_\inlinecode{left}$}] Unary operator expressions.
	\item[PrintNode \inlinecode{print reg$_\inlinecode{1}$}] Print statements.
	\item[GlobalNode \inlinecode{global reg$_\inlinecode{1}$}] Global statements.
%	\item[AssertInterable] Used to assert that an object is iterable and has a given length.
%	\item[ImportNode \inlinecode{import reg$_\inlinecode{1}$}] Introduced when the import statement is in the AST.
	\item[AssertNode] Assertion for path sensivity.
\end{description}
